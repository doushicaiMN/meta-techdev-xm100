144 //! Open a serial port
145 /*!
146  \param name Symbolic link name for port
147  \return handle to port
148 */
149 HANDLE COM_OPEN(const CH* name)
150 {
151 #ifdef WIN32
152  HANDLE h;
153  DCB dcb;
154  COMMTIMEOUTS cto;
155  h = CreateFile(name, GENERIC_READ | GENERIC_WRITE, 0, NULL, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, NULL);
156  if (h != INVALID_HANDLE_VALUE)
157  {
158  memset(&cto, 0, sizeof(cto));
159  cto.ReadIntervalTimeout = MAXDWORD;
160  memset(&dcb, 0, sizeof(dcb));
161  dcb.DCBlength = sizeof(dcb);
162  dcb.fDtrControl = 1;
163  dcb.fRtsControl = 1;
164  if (!SetupComm(h, 4096, 32768))
165  MESSAGE(MSG_DBG, "SetupComm err=%d", GetLastError());
166  if (!SetCommTimeouts(h, &cto))
167  MESSAGE(MSG_DBG, "SetCommTimeouts err=%d", GetLastError());
168  if (!BuildCommDCB("baud=9600 parity=N data=8 stop=1", &dcb))
169  MESSAGE(MSG_DBG, "BuildCommDCB err=%d", GetLastError());
170  if (!SetCommState(h, &dcb))
171  MESSAGE(MSG_DBG, "SetCommState err=%d", GetLastError());
172  return h;
173  }
174  return NULL;
175 #else
176  // open the serial port with the NDELAY option
177  // otherwise this may block forever because u-blox chips
178  // don't support the DCD line
179  int fd = open(name, O_RDWR | O_NOCTTY | O_NDELAY);
180  if (fd < 0)
181  {
182  return (HANDLE)0;
183  }
184 
185  // remove the NDELAY option again, it will be
186  // handled on the lower layer when setting the baudrate
187  fcntl(fd, F_SETFL, 0);

192 //! set baudrate of serial COM port
193 /*!
194  \param h Handle to device
195  \param br Baudrate
196  \return #TRUE if success, #FALSE else
197 */
198 BOOL COM_BAUDRATE(HANDLE h, U4 br)
199 {
200 #ifdef WIN32
201  DCB dcb;
202  dcb.DCBlength = sizeof(dcb);
203  if (!GetCommState(h, &dcb))
204  {

206  return FALSE;

209  dcb.ByteSize = 8;
210  dcb.fBinary = TRUE;
211  dcb.DCBlength = sizeof(dcb);
212  if (!SetCommState(h, &dcb))
213  {
214  MESSAGE(MSG_DBG, "SetCommState err=%d", GetLastError());
215  return FALSE;
216  }
217  return TRUE;
218 #else
219  struct termios tp;
220  int baudrate;
221 
222  switch(br)
223  {
224 #if !defined(__APPLE__)
225  case 921600:baudrate=B921600;break;
226  case 460800:baudrate=B460800;break;
227 #endif
228  case 230400:baudrate=B230400;break;
229  case 115200:baudrate=B115200;break;
230  case 57600: baudrate=B57600;break;
231  case 38400: baudrate=B38400;break;
232  case 19200: baudrate=B19200;break;
233  case 9600: baudrate=B9600;break;
234  default: baudrate=B9600;
235  }
236 
237  tcgetattr((int)h, &tp);
238 
239  //cfmakeraw(&tp);
240  tp.c_lflag = 0;
241  tp.c_iflag = 0;
242  tp.c_oflag = 0;

244  tp.c_cflag |= CLOCAL | CS8;

254  cfsetospeed(&tp, baudrate);

256

259  return FALSE;
260  }
261  return TRUE;
262 #endif
263 }

483 //! write data to I2C port
484 /*!
485  \param h handle to device
486  \param deviceAddress address of the I2C device
487  \param p pointer to data to write
488  \param size size of data to write
489  \return number of bytes written
490 */
491 U4 U2C_WRITE(HANDLE h, int deviceAddress, const void* p, U4 size)
492 {
493  U4 writtenBytes = 0;
494  U2C_TRANSACTION trans;
495  trans.nMemoryAddress = 0;
496  trans.nMemoryAddressLength = 0;
497  trans.nSlaveDeviceAddress = deviceAddress;
498 
499  while (size)
500  {
501  U2 sizeTrans = (U2)MIN(size, sizeof(trans.Buffer));
502  trans.nBufferLength = sizeTrans;
503  memcpy(trans.Buffer,p,trans.nBufferLength);
504  U2C_RESULT res = U2C_Write(h,&trans);
505  if (res != U2C_SUCCESS)
506  {
507  MESSAGE(MSG_ERR, "U2C write failed: %s (%d)", U2C_StatusString(res), res);
508  break;
509  }
510  writtenBytes += sizeTrans;
511  size -= sizeTrans;
512  p = (const void*)((U1*)p+sizeTrans);
513  }
514  return (U4)writtenBytes;
515 }

657 //! write data to SPI port
658 /*!
659  \param h handle to device
660  \param pReadBuf pointer to device buffer structure
661  \param p pointer to data to write
662  \param size size of data to write
663  \return number of bytes written
664 */
665 U4 SPU_WRITE(HANDLE h, SPI_READBUFFER_pt pReadBuf, const void* p, U4 size)
666 {
667  // manage read buffer. as SPI always performs a read access and a write access at the
668  // same time, we temporarily store the data read and return it on an SPI_READ call
669  U4 room = sizeof(pReadBuf->buffer)-(pReadBuf->size);
670  U4 transfSize = MIN(room, size);
671  if (!transfSize)
672  {
673  return 0;
674  }
675 
676  U2C_SingleIoWrite(h, 5, FALSE); // slave select
677  U4 alreadyTransferred = 0;
678  while (transfSize)
679  {
680  // Diolan allows single-transfers of no more than 256 byte
681  U2 thisSize = (U2)MIN(transfSize, 256);
682 
683  // perform write/read
684  U2C_RESULT res = U2C_SpiReadWrite(h, // handle
685  (U1*)p+alreadyTransferred, // write buffer
686  (BYTE*)pReadBuf->pBuffer + pReadBuf->size, // read buffer pointer
687  (U2)thisSize); // r/w size
688 
689  if (res != U2C_SUCCESS)
690  {
691  MESSAGE(MSG_ERR, "SPI write data: %s (%d)", U2C_StatusString(res), res);
692  U2C_SingleIoWrite(h,5,TRUE); // slave deselect
693  return alreadyTransferred;
694  }
695 
696  pReadBuf->size += thisSize;
697  alreadyTransferred += thisSize;
698  transfSize -= thisSize;
699 
700  }
701  U2C_SingleIoWrite(h, 5, TRUE); // slave deselect
702  return (U4)size;
703 }

761  }
762  }
763  }
764  else
765  {
766  // we received only FFs -> discard
767  return 0;
768  }
769  }
770

1543  const void *p,
1544  U4 size)
1545 {
1546  // get the header of the UBX message
1547  UBX_HEAD_pt pUbxHdr = (UBX_HEAD_pt)p;
1548 
1549  // check for UPD-FLWRI message
1550  if(pUbxHdr->classId == UBX_CLASS_UPD && pUbxHdr->msgId == UBX_UPD_FLWRI)
1551  {
1552  // content of UPD-FLWRI message
1553  U4 targetAddress = 0; // address where to write the data to
1554  U4 writeSize = 0; // length of the payload to be written
1555 
1556  U4 startOfData = 0; // where the data section starts
1557  U4 pos = 0; // current position inside the complete packet
1558  U1 *pData = NULL; // pointer to data section of packet
1559  U1 totalFF = 8; // how many 0xFF in success are allowed
1560  U curSearch = 0; // how man 0xFF have been already found
1561 
1562  // get the content of the UPD-FLWRI message
1563  memcpy(&targetAddress, (U4 *)((U1 *)p+sizeof(UBX_HEAD_t)), sizeof(targetAddress));
1564  memcpy(&writeSize, (U4 *)((U1 *)p+sizeof(UBX_HEAD_t)+sizeof(targetAddress)), sizeof(writeSize));
1565 
1566  // set the pointer to the data
1567  pData = (U1 *)p + sizeof(UBX_HEAD_t) + sizeof(targetAddress) + sizeof(writeSize);
1568 
1569  while(pos < writeSize)
1570  {
1571  // get the current character
1572  U1 current = pData[pos];
1573 
1574  // compare to 0xFF
1575  if(current == 0xFF)
1576  {
1577  // increment the 0xFF counter
1578  curSearch++;
1579  }
1580  // something else found
1581  else
1582  {
1583  // check if already more 0xFF have passed than allowed
1584  if(curSearch > totalFF)
1585  {
1586  // check if anything has to be written
1587  if(pos-startOfData-curSearch)
1588  {

1590  U4 writeAddress = targetAddress+startOfData;
1591  U4 newWriteSize = pos-startOfData-curSearch;

1621  break;
