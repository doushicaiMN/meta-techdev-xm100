51 U4 ValidateImage(IN FWHEADER_t const *pImage,
52  IN const size_t fileSize,
53  OUT FWFOOTERINFO_t *pFwFooter)
54 {
55  CH verString[100];
56  //check file size, must be at least header size
57  if (fileSize < sizeof(FWHEADER_t))
58  {
59  return 0;
60  }
61  //get the header
62  U4 generation =
63  (pImage->v1.magic == magicG50) ? 50 : // u-blox5
64  (pImage->v1.magic == magicG51) ? 60 : // u-blox6
65  (pImage->v1.magic == magicG70) ? 70 : // u-blox7
66  (pImage->v1.magic == magicG80) ? 80 : // u-blox8
67  0;
68  /***************************************************
69  * Check if valid image footer is present and parse *
70  ***************************************************/
71  {
72  CH* pMagic;
73  BOOL magicFound = FALSE;
74  U i;
75  for (i = 0; i < 4; i++) // image might be padded with up to 3 0xFF
76  {
77  pMagic = ((CH*)pImage) + fileSize - 4 - i;
78  if (pMagic[0] == 'U' && pMagic[1] == 'B' && pMagic[2] == 'F' && pMagic[3] == 'L')
79  {
80  magicFound = TRUE;
81  break;
82  }
83  }
84  if (magicFound)
85  {
86  U4 version;
87  U4 crc;
88  U4 footerSize;
89  memcpy(&version, pMagic - sizeof(U4), sizeof(version));
90  memcpy(&pFwFooter->numberOfImages, pMagic - (3 * sizeof(U4)), sizeof(version));
91  footerSize = 5 * sizeof(U4) + sizeof(U4) * pFwFooter->numberOfImages;
92  if (version == 0 &&
93  pFwFooter->numberOfImages <= NUMOF(pFwFooter->imagesSize) &&
94  memcpy(&crc, pMagic - (footerSize - sizeof(U4)), sizeof(crc)) != 0 &&
95  lib_crc_crc32(0xF5D4C69, pMagic - (footerSize - 2 * sizeof(U4)), (footerSize - sizeof(U4))) == crc)
96  {
97  for (i = 0; i < pFwFooter->numberOfImages; i++)
98  {
99  memcpy(&pFwFooter->imagesSize[i], pMagic - (footerSize - (sizeof(U4) * (i + 2))), sizeof(pFwFooter->imagesSize[i]));
100  MESSAGE(MSG_DBG, "image %u size %u", i, pFwFooter->imagesSize[i]);
101  }
102  memcpy(&pFwFooter->cfgSize, pMagic - 2 * sizeof(U4), sizeof(pFwFooter->cfgSize));
103  MESSAGE(MSG_DBG, "config size %u", pFwFooter->cfgSize);
104  if (generation == 0)
105  {
106  return 91; //Image magic is not present but valid version 0 image footer found so this is U-blox 9 image, cannot check anything more because it is scrumbled
107  }
108  }
109  else
110  {
111  MESSAGE(MSG_ERR, "Invalid footer found");
112  return 0;
113  }
114  }
115  }
116  if (!generation)
117  {
118  // magic word not found
119  MESSAGE(MSG_LEV2, "No magic word found");
120  return 0;
121  }
122 
123  if (pImage->v1.pEnd < pImage->v1.pStart ||
124  pImage->v1.pStart < pImage->v1.pBase ||
125  ((U4)pImage->v1.pEnd & 0xFF800000) != EXT2_BASE ||
126  ((U4)pImage->v1.pBase & 0xFF800000) != EXT2_BASE ||
127  ((U4)pImage->v1.pStart & 0xFF800000) != EXT2_BASE )
128  {
129  //pImage->pointer(s) seems to be corrupt
130  MESSAGE(MSG_LEV2, "Image addresses invalid");
131  return FALSE;
132  }
133  //magic word found and pointers seem to be OK, check CRC
134  U4 crcpos = (pImage->v1.pEnd & ~0x1) - pImage->v1.pBase;
135  U4 crcrange = crcpos - 4; // CRC starts behind magic word
136  BOOL crcOk = CheckUbxChecksumU4((U4*)((U1*)pImage+4), crcrange);
137 
138  // one could return at this stage to detect validity of the image. we
139  // additionally grab the version string out of the image
140  // the ROM version the firmware was built for is contained in the lower
141  // 7 bits of the highest byte of the hardware info word
142  if (crcOk)
143  {
144  memset(verString, 0, sizeof(verString));
145  if ((pImage->v1.pVersion & 0xFF800000) == EXT2_BASE)
146  {
147  CH* FwVerStr = (CH*)((U1*)pImage + (pImage->v1.pVersion - pImage->v1.pBase));
148  if (FwVerStr < (CH*)((U1*)pImage + fileSize))
149  {
150  strncpy(verString, FwVerStr, sizeof(verString));
151  }
152  }
153 
154  MESSAGE(MSG_LEV2, "Image (file size %u) for u-blox%d accepted", fileSize, generation / 10);
155  MESSAGE(MSG_LEV2, "Image Ver '%s'", verString);
156  MESSAGE(MSG_DBG, "CRC= 0x%08X 0x%08X",
157  *((U4*)((U1*)pImage + crcpos)),
158  *((U4*)((U1*)pImage + crcpos + 4)));
159  }
160  return crcOk ? generation : 0;
161 }

165 BOOL OpenAndBufferFile(IN const CH* BinaryFileName,
166  OUT FWHEADER_t** ppFileContent,
167  OUT size_t* pFileSize)
168 {
169  if (!BinaryFileName || !strlen(BinaryFileName))
170  {
171  MESSAGE(MSG_ERR, "File '%s' is not valid", BinaryFileName);
172  return FALSE;
173  }
174 
175  FILE *stream=fopen(BinaryFileName, "rb");
176 
177  if (!stream)
178  {
179  MESSAGE(MSG_ERR, "Could not open file '%s'.", BinaryFileName);
180  return FALSE;
181  }
182 
183  //get filesize and allocate memory
184  int ret=fseek(stream, 0, SEEK_END);
185  if(ret)
186  {
187  MESSAGE(MSG_ERR, "Could not set pointer to start in file '%s'", BinaryFileName);
188  fclose(stream);
189  return FALSE;
190  }
191  long filesize = ftell(stream);
192  long ffAppend = (4 - (filesize & 0x3)) & 0x3;
193  CH* fileTmp = (CH*)malloc(filesize + ffAppend);
194  if (fileTmp == NULL)
195  {
196  MESSAGE(MSG_ERR, "Allocation of file buffer memory failed.");
197  fclose(stream);
198  return FALSE;
199  }
200 
201  //copy file content to ram buffer
202  ret=fseek(stream, 0, SEEK_SET);
203  if(ret)
204  {
205  MESSAGE(MSG_ERR, "Seeking in the file failed.");
206  fclose(stream);
207  free(fileTmp);
208  return FALSE;
209  }
210 
211  // Read data
212  size_t numRead=fread(fileTmp, 1, (size_t)filesize, stream);
213  fclose(stream);
214  if(numRead!=(size_t)filesize)
215  {
216  MESSAGE(MSG_ERR, "Wrong number of bytes read. Read = %ld, Expected = %ld", numRead, filesize);
217  free(fileTmp);
218  return FALSE;
219  }
220  memset(fileTmp + filesize, 0xFF, ffAppend); // append FFs
221  *pFileSize = filesize + ffAppend;
222  *ppFileContent = (FWHEADER_t*)fileTmp;
223  return TRUE;
224 }
