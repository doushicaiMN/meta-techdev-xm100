143  , IN U1 classId
144  , IN U1 msgId
145  , IN CH* payload
146  , IN U4 payloadSize )
147 {
148  assert(rcv);
149  assert( ( payload && payloadSize)

151 
152  CH* pMessage;

154  if (!UbxCreateMessage( classId

170 UBX_HEAD_t* rcvPollMessage( INOUT RCV_DATA_t *rcv
171  , IN U1 classId
172  , IN U1 msgId
173  , IN CH* payload
174  , IN U4 payloadSize
175  , IN U4 timeout)
176 {
177  assert(rcv);
178  assert( ( payload && payloadSize)
179  || (!payload && !payloadSize));
180 
181  U4 retryCount;
182  for(retryCount = 0; retryCount < RETRY_COUNT; retryCount++)
183  {
184  UBX_HEAD_t* msg = rcvPollMessageOnce(rcv, classId, msgId, payload, payloadSize, timeout);
185  if(msg != NULL)
186  return msg;
187 
188  MESSAGE(MSG_DBG, "Retry poll");
189  }
190  return NULL;

193 int rcvAckMessage( INOUT RCV_DATA_t *rcv
194  , IN U1 classId
195  , IN U1 msgId
196  , IN CH* payload
197  , IN U4 payloadSize
198  , IN U4 timeout )
199 {
200  assert(rcv);
201  assert( ( payload && payloadSize)
202  || (!payload && !payloadSize));
203 
204  U4 retryCount;
205  for(retryCount = 0; retryCount < RETRY_COUNT; retryCount++)
206  {
207  if( rcvSendMessage(rcv, classId, msgId, payload, payloadSize) )
208  {
209  const U4 toTime = TIME_GET() + timeout;
210  do // loop so message is not retransmitted when ACK from different (previous) message is already in buffor or is received.
211  {
212  UBX_HEAD_t *ack = rcvReceiveMessage(rcv, toTime - TIME_GET(), UBX_CLASS_ACK, -1);
213  if (ack != NULL)
214  {
215  UBX_ACK_ACK_t *pAck = (UBX_ACK_ACK_t*)((U1*)ack + UBX_HEAD_SIZE);
216  if (pAck->clsId == classId &&
217  pAck->msgId == msgId)
218  {
219  BOOL success = ack->msgId == UBX_ACK_ACK;
220  free(ack);
221  return success ? 1 : 0;
222  }
223  free(ack);
224  }
225  }
226  while (TIME_GET() < toTime);
227  }
228  }
229  return -1;
230 }

232 UBX_HEAD_t* rcvReceiveMessage(INOUT RCV_DATA_t *rcv, IN U4 timeout, IN I4 classId, IN I4 msgId)
233 {
234  assert(rcv);
235 
236  const U4 toTime = TIME_GET() + timeout;
237  do
238  {
239  U4 availableSize = RECEIVEBUF_SIZE - (rcv->mRecBuf.pEnd - rcv->mRecBuf.pCurrent);
240  U4 received = SER_READ(rcv->mPortHandle, rcv->mRecBuf.pEnd, availableSize);
241 
242  rcv->mRecBuf.pEnd += received;
243  U1* pMessageBegin = NULL;
244 
245  //parse buffer if a valid UBX message can be found
246  //start at position pCurrent
247  if (UbxSearchMsg(rcv->mRecBuf.pCurrent, rcv->mRecBuf.pEnd - rcv->mRecBuf.pCurrent, &pMessageBegin))
248  {
249  UBX_HEAD_t ubxhead;
250  memcpy(&ubxhead, pMessageBegin, sizeof(ubxhead));
251  CH* message = (CH*)malloc((ubxhead.size + UBX_FRAME_SIZE)*sizeof(CH));
252  if (!message)
253  {
254  return NULL;
255  }
256 
257  // copy the message to the buffer
258  memcpy(message, pMessageBegin, ubxhead.size + UBX_FRAME_SIZE);
259 
260  rcv->mRecBuf.pCurrent = pMessageBegin + ubxhead.size + UBX_FRAME_SIZE;
261 
262  // move the rest of the data to the beginning of the buffer
263  memmove(rcv->mRecBuf.Buf, rcv->mRecBuf.pCurrent, rcv->mRecBuf.pEnd - rcv->mRecBuf.pCurrent);
264  rcv->mRecBuf.pEnd -= rcv->mRecBuf.pCurrent - rcv->mRecBuf.Buf;
265  rcv->mRecBuf.pCurrent = rcv->mRecBuf.Buf;
266 
267  if ((classId == -1 || classId == ubxhead.classId)
268  && (msgId == -1 || msgId == ubxhead.msgId))
269  {
270  return (UBX_HEAD_t*)message;
271  }
272  free(message);
273  }
274  else
275  {
276  TIME_SLEEP(1); // don't loop at 100% CPU
277  }
278  }
279  while(TIME_GET() < toTime);
280 
281  return NULL;
282 }

304 UBX_HEAD_t* rcvDoAutobaud(INOUT RCV_DATA_t *rcv, IN BOOL sendTraining)
305 {
306  assert(rcv);
307  U4 retryCount = 0;
308  U4 autoBaudIndex = 0;
309 
310  for(;;)
311  {
312  UBX_HEAD_t* msg = rcvPollMessage(rcv, UBX_CLASS_MON, UBX_MON_VER, NULL, 0, AUTOBAUD_TIMEOUT);
313  if(msg != NULL)
314  {
315  MESSAGE(MSG_DBG,"Detected Baudrate is %d", rcv->mPortHandle->baudrate);
316  return msg;
317  }
318 
319  if (retryCount++ <= RETRY_COUNT_AUTOBAUD)
320  {
321  MESSAGE(MSG_DBG,"...retrying autobaud");
322  }
323  else if(autoBaudIndex < NUMOF(gAutoBaudRates))
324  {
325  // advance to the next baudrate
326  MESSAGE(MSG_DBG, "Retrying with baudrate %d", gAutoBaudRates[autoBaudIndex]);
327  rcvSetBaud(rcv, gAutoBaudRates[autoBaudIndex]);
328 
329  // send the training sequence
330  if(sendTraining)
331  rcvSendTrainingSequence(rcv);
332 
333  ++autoBaudIndex;
334  retryCount = 0;
335  }
336  else
337  {
338  MESSAGE(MSG_ERR, "Unable to Communicate on any Baudrate");
339  return NULL;
340  }
341  }
342 }
