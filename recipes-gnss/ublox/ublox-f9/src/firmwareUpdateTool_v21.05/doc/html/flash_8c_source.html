169 I4 GetSectorNrForSize(IN const U4 Offset,
170  IN const U4 FwSize,
171  IN const BLOCK_ARR_pt pFlashdef)
172 {
173  assert(pFlashdef);
174  U4 size = 0; //size
175  U4 block = 0; //actual block
176  U4 blocksec = 0; //sector in the actual block
177  U4 sectorNr = 0; //count sector numbers
178 
179  U4 addr = 0;
180  while (addr < Offset)
181  {
182  addr += pFlashdef->pBlocks[block].Size;
183  blocksec++;
184  if (blocksec >= pFlashdef->pBlocks[block].Count)
185  {
186  block++;
187  blocksec = 0;
188  if (block >= pFlashdef->NumBlocks)
189  {
190  //there are no more sectors behind size of flash
191  return -1;
192  }
193  }
194  }
195 
196  while (size < FwSize)
197  {
198  if ((block >= pFlashdef->NumBlocks) && (size < FwSize))
199  {
200  //somewhat went wrong
201  MESSAGE(MSG_ERR, "Cannot determine packet nr for size!");
202  return -1;
203  }
204  sectorNr ++;
205  size += pFlashdef->pBlocks[block].Size;
206  blocksec++;
207  if (blocksec >= pFlashdef->pBlocks[block].Count)
208  {
209  block++;
210  blocksec = 0;
211  }
212  }
213  return sectorNr;
214 }

216 I4 GetPacketNrForSector(IN const U4 Sector,
217  IN const BLOCK_ARR_pt pFlashdef,
218  IN const U4 PacketSize)
219 {
220  assert(pFlashdef);
221  if(pFlashdef->NumBlocks == 1)
222  {
223  // uniform flash
224  return Sector * pFlashdef->pBlocks[0].Size / PacketSize;
225  }
226  else
227  {
228  I4 packet = 0;
229  U4 block = 0; //actual block
230  U4 blocksec = 0; //sector in the actual block
231 
232  U4 sec_cnt = 0;
233  // sum up the number of packets of the sectors
234  while (sec_cnt < Sector)
235  {
236  sec_cnt++;
237  if (block >= pFlashdef->NumBlocks)
238  {
239  //somewhat went wrong
240  MESSAGE(MSG_ERR, "GetPacketNrForSector: Sec %u Packetsize %u", Sector, PacketSize);
241  return -1;
242  }
243  packet += pFlashdef->pBlocks[block].Size / PacketSize;
244  blocksec++;
245  if (blocksec >= pFlashdef->pBlocks[block].Count)
246  {
247  block++;
248  blocksec = 0;
249  }
250  }
251  return packet;
252  }
253 }

255 I4 GetSectorNrForAddress(IN const U4 Address,
256  IN const U4 Base,
257  IN const U4 Offset,
258  IN const BLOCK_ARR_pt pFlashdef)
259 {
260  assert(pFlashdef);
261  U4 block = 0; //actual block
262  U4 blocksec = 0; //sector in the actual block
263 
264  // advance block and blocksec to offset address
265  U4 addr = 0;
266  while (addr < Offset)
267  {
268  addr += pFlashdef->pBlocks[block].Size;
269  blocksec++;
270  if (blocksec >= pFlashdef->pBlocks[block].Count)
271  {
272  //next block
273  block++;
274  blocksec = 0;
275  if (block >= pFlashdef->NumBlocks)
276  {
277  //cannot erase a section behind size of flash
278  return -1;
279  }
280  }
281  }
282 
283  // sum up the number of sectors
284  U4 sec_cnt = 0;
285  while (addr < (Address-Base+Offset))
286  {
287  addr += pFlashdef->pBlocks[block].Size;
288  // address within the limits of the actual sector?
289  if (addr > (Address-Base+Offset))
290  {
291  break;
292  }
293  // perhaps address fits into next sector's ranges
294  sec_cnt ++;
295  blocksec++;
296  if (blocksec >= pFlashdef->pBlocks[block].Count)
297  {
298  block++;
299  blocksec = 0;
300  if (block >= pFlashdef->NumBlocks)
301  {
302  // there are no more sectors available on flash
303  return -1;
304  }
305  }
306  }
307  return sec_cnt;
308 }

331 BOOL GetFlashOrganisation(IN U2 const ManId,
332  IN U2 const DevId,
333  INOUT BLOCK_ARR_pt pBlocks,
334  INOUT U4* const pSize,
335  IN CH const * const Filename)
336 {
337  assert(pBlocks);
338  CH* file = NULL;
339  CH* fileTmp = NULL;
340  if (strlen(Filename)>0) // is there a file at all?
341  {
342  FILE *stream=fopen(Filename, "r");
343  if (!stream)
344  {
345  //MESSAGE(MSG_ERR, "Could not open file '%s'", Filename);
346  return FALSE;
347  }
348 
349  //clear flash definition description vector
350  clearBlocks(pBlocks);
351 
352  //get file size and allocate memory
353  int ret=fseek(stream, 0, SEEK_END);
354  if(ret)
355  {
356  fclose(stream);
357  return FALSE;
358  }
359  long filesize = ftell(stream);
360  fileTmp = (CH*) malloc(filesize);
361  if (fileTmp == NULL)
362  {
363  MESSAGE(MSG_ERR, "Allocation of file buffer memory failed.");
364  fclose(stream);
365  return FALSE;
366  }
367  //copy file content to ram buffer
368  rewind(stream);
369  size_t numRead=fread(fileTmp, 1, filesize, stream);
370  fclose(stream);
371  if(numRead!=(size_t)filesize)
372  {
373  free(fileTmp);
374  return FALSE;
375  }
376  file = fileTmp;
377  }
378  else
379  {
380  // use compiled-in version
381  MESSAGE(MSG_DBG,"Use compiled-in Flash Description");
382  fileTmp = (CH*) malloc(sizeof(flashTxt));
383  if (fileTmp == NULL)
384  {
385  MESSAGE(MSG_ERR, "Allocation of file buffer memory failed.");
386  return FALSE;
387  }
388  memcpy(fileTmp, flashTxt, sizeof(flashTxt));
389  file = fileTmp;
390  }
391 
392  //file content now in 'file', process it
393  CH* pEnd;
394  CH key[32];
395  //create search string
396  sprintf(key, "0x%04X-0x%04X", ManId, DevId);
397  //search for manid, devid
398  CH* pPos = strstr(file, key);
399  if (pPos == NULL)
400  {
401  free(file);
402  return FALSE;
403  }
404  //description was found, extract organization
405  pPos = strchr(pPos, '=');
406  if (pPos == NULL)
407  {
408  free(file);
409  return FALSE;
410  }
411  // remove equal sign and all leading spaces
412  pPos += strspn(pPos, "\t =");
413  // remove comments
414  pEnd = strpbrk(pPos, ";\r\n\0");
415  if(pEnd == NULL)
416  {
417  free(file);
418  return FALSE;
419  }
420  *pEnd = '\0';
421  // remove trailing spaces
422  while ((pEnd >= pPos) && (isspace(*pEnd) || (*pEnd == 0) || (*pEnd == '\n') || (*pEnd == '\r')))
423  {
424  *pEnd = '\0';
425  pEnd --;
426  }
427  // parse syntax
428  *pSize = 0;
429  while (*pPos != '\0')
430  {
431  CH* stopstring;
432  U4 value = strtoul(pPos, &stopstring, 10);
433  if ((stopstring > pPos) && (*stopstring == 'x') && (value > 0))
434  {
435  BLOCK_DEF_t block;
436  block.Count = value;
437  pPos = stopstring + 1;
438  value = strtoul(pPos, &stopstring, 10);
439  if ((stopstring > pPos) && (value > 0))
440  {
441  block.Size = value * 1024;
442  pPos = stopstring;
443 
444  *pSize += block.Count * block.Size;
445  addBlock(pBlocks, &block);
446  //skip spaces
447  while (isspace(*pPos) || *pPos == ',')
448  {
449  pPos ++;
450  }
451  }
452  }
453  }
454  if (fileTmp)
455  free(fileTmp);
456 
457  return (pBlocks->NumBlocks != 0);
458 }

470 BOOL addBlock(INOUT BLOCK_ARR_pt pBlocks,
471  IN BLOCK_DEF_pt const pNewBlock)
472 {
473  assert(pBlocks && pNewBlock);
474  assert( ( pBlocks->pBlocks && pBlocks->NumBlocks)
475  || (!pBlocks->pBlocks && !pBlocks->NumBlocks) );
476 
477  BOOL result=FALSE;
478  BLOCK_DEF_pt pTmpBlocks = (BLOCK_DEF_pt) realloc(pBlocks->pBlocks, (size_t)((pBlocks->NumBlocks + 1)*sizeof(*pNewBlock)));
479 
480  if(pTmpBlocks)
481  {
482  memcpy(&pTmpBlocks[pBlocks->NumBlocks], pNewBlock, sizeof(*pNewBlock));
483  pBlocks->pBlocks=pTmpBlocks;
484  ++pBlocks->NumBlocks;
485  result=TRUE;
486  }
487 
488  return result;
489 }
